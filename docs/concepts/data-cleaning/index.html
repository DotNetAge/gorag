    数据清洗 (Data Cleaning)¶

    什么是数据清洗？¶

    数据清洗是 RAG 系统预处理的关键环节，负责对解析后的纯文本进行去噪、标准化、格式转换等处理，确保后续分块和向量化的质量。

    一句话理解¶

    数据清洗 = 把"脏乱差"的文本变成"干净整洁"的结构化文本

    数据清洗在 RAG 中的位置¶

    flowchart LR A[原始文档] --> B[文档解析] B --> C[纯文本] C --> D[数据清洗] D --> E[清洗后文本] E --> F[分块处理] F --> G[向量化] style D fill:#fff3e0 style E fill:#e8f5e9

    为什么需要数据清洗？¶

    原始文本的常见问题¶

    问题类型 示例 对 RAG 的影响
    HTML 残留 &nbsp;, <br>, &amp; 噪音字符干扰
    特殊符号 多余空格、制表符、换行符 分块边界错误
    大小写混乱 Title / TITLE / title 关键词匹配问题
    超链接残留 [点击这里](http://example.com) 干扰理解
    代码片段 行号、缩进混乱 语义混淆
    水印/页眉页脚 "机密文件"、页码 引入无关信息
    语言混杂 中英混杂、表情符号 向量化效果下降

    数据清洗的价值¶

    清洗前："在这个炎热的夏天   ，我们去游泳。\n\n\n\n\n      "
清洗后："在这个炎热的夏天，我们去游泳。"

    数据清洗核心操作¶

    1. 去除噪音字符¶

    常见的噪音字符¶

    类型 示例 清洗方法
    HTML 实体 &nbsp;, &lt;, &gt; HTML 实体解码
    特殊空白 \u200b (零宽空格), \xa0 (不换行空格) Unicode 规范化
    控制字符 \x00, \x08 正则过滤
    多余空白 连续空格、Tab、换行 规范化为空格 
    // 去除噪音字符
func CleanNoiseText(text string) string {
    // 1. 解码 HTML 实体
    text = html.UnescapeString(text)

    // 2. 规范化 Unicode
    text = strings.ToLower(text)

    // 3. 替换特殊空白字符为空格
    text = regexp.MustCompile(`[\x00-\x08\x0b\x0c\x0e-\x1f\x7f]`).ReplaceAllString(text, "")

    // 4. 规范化空白字符
    text = regexp.MustCompile(`[\t\p{Zs}]+`).ReplaceAllString(text, " ")

    // 5. 去除零宽字符
    text = regexp.MustCompile(`[\u200b-\u200f\u2028-\u202f\ufeff]`).ReplaceAllString(text, "")

    return strings.TrimSpace(text)
}

    2. 规范化处理¶

    2.1 繁简体转换¶

    // 繁简体转换
func NormalizeChinese(text string) string {
    // 使用 opencc 或 goregen 库
    simplified, _ := opencc.Convert(text, "t2s")
    return simplified
}

    2.2 全角半角转换¶

    // 全角转半角
func FullwidthToHalfwidth(text string) string {
    var result []rune
    for _, r := range text {
        // 全角字符范围：0xFF01-0xFF5E
        if r >= 0xFF01 && r <= 0xFF5E {
            result = append(result, r-0xFEE0)
        } else if r == 0x3000 {
            // 全角空格
            result = append(result, 0x0020)
        } else {
            result = append(result, r)
        }
    }
    return string(result)
}

    2.3 大小写规范化¶

    场景 处理方式 示例
    英文全大写 转小写 PDF → pdf
    标题大小写 保持原样 Title
    专有名词 保持原样 iPhone, macOS

    3. 去除超链接和格式残留¶

    // 去除 Markdown/HTML 链接
func RemoveLinks(text string) string {
    // 去除 Markdown 链接 [text](url)
    text = regexp.MustCompile(`\[([^\]]+)\]\([^\)]+\)`).ReplaceAllString(text, "$1")

    // 去除 Markdown 图片 ![alt](url)
    text = regexp.MustCompile(`!\[[^\]]*\]\([^\)]+\)`).ReplaceAllString(text, "")

    // 去除 HTML 链接 <a href="...">text</a>
    text = regexp.MustCompile(`<a[^>]*>([^<]*)</a>`).ReplaceAllString(text, "$1")

    // 去除裸露的 URL
    text = regexp.MustCompile(`https?://[^\s]+`).ReplaceAllString(text, "")

    return text
}

    4. 去除代码特殊元素¶

    // 去除代码中的行号
func RemoveCodeLineNumbers(text string) string {
    // 去除行号 "    1  code" -> "code"
    text = regexp.MustCompile(`^\s*\d+\s+`).ReplaceAllString(text, "")

    // 去除代码缩进指示符（> 用于引用）
    text = regexp.MustCompile(`^>\s*`).ReplaceAllString(text, "")

    return text
}

    5. 去除水印和页眉页脚¶

    企业文档中常见的"机密"、"内部使用"等水印需要识别和去除。

    // 常见水印关键词
var watermarkPatterns = []string{
    "机密", "CONFIDENTIAL", "内部文件", "INTERNAL",
    "版权所有", "copyright", "未经授权",
}

func RemoveWatermarks(text string) string {
    for _, pattern := range watermarkPatterns {
        text = regexp.MustCompile(`(?i)` + pattern).ReplaceAllString(text, "")
    }
    return text
}

    分块边界清洗¶

    分块前的文本清洗对分块质量至关重要。

    1. 规范化段落分隔¶

    // 规范化段落分隔
func NormalizeParagraphSeparation(text string) string {
    // 将多个连续换行符规范化为两个（段落分隔）
    text = regexp.MustCompile(`\n{3,}`).ReplaceAllString(text, "\n\n")

    // 去除段首段尾多余空格
    lines := strings.Split(text, "\n")
    for i, line := range lines {
        lines[i] = strings.TrimSpace(line)
    }

    return strings.Join(lines, "\n")
}

    2. 句子边界识别¶

    句子边界识别对于语义分块很重要。

    // 句子边界检测
func SplitSentences(text string) []string {
    // 常见句子结束符：. ! ? ；。！？""
    // 但需要排除缩写、小数点等

    sentences := make([]string, 0)
    runes := []rune(text)
    var current []rune

    for i := 0; i < len(runes); i++ {
        current = append(current, runes[i])

        if isSentenceBoundary(runes, i) {
            sentences = append(sentences, strings.TrimSpace(string(current)))
            current = nil
        }
    }

    if len(current) > 0 {
        sentences = append(sentences, strings.TrimSpace(string(current)))
    }

    return sentences
}

func isSentenceBoundary(runes []rune, i int) bool {
    switch runes[i] {
    case '。', '！', '？', '｡':
        return true
    case '.', '!', '?':
        // 排除缩写（如 U.S.A.）和小数点
        if i+1 < len(runes) && runes[i+1] >= 'A' && runes[i+1] <= 'Z' {
            return false
        }
        if i > 0 && runes[i-1] >= '0' && runes[i-1] <= '9' {
            return false
        }
        return true
    }
    return false
}

    数据清洗流水线¶

    将多个清洗步骤组合成流水线：

    // 数据清洗流水线
type CleaningPipeline struct {
    steps []CleaningStep
}

type CleaningStep func(string) string

func NewCleaningPipeline() *CleaningPipeline {
    return &CleaningPipeline{
        steps: []CleaningStep{
            CleanNoiseText,
            RemoveLinks,
            NormalizeParagraphSeparation,
            FullwidthToHalfwidth,
            NormalizeChinese,
            RemoveWatermarks,
        },
    }
}

func (p *CleaningPipeline) Execute(text string) string {
    result := text
    for _, step := range p.steps {
        result = step(result)
    }
    return result
}

// 使用示例
func main() {
    pipeline := NewCleaningPipeline()

    rawText := "这是   一个   文档。<br>包含&amp;符号   和链接 [点击](http://example.com)"

    cleanText := pipeline.Execute(rawText)
    // 输出: "这是 一个 文档。包含&符号 和链接 点击"
}

    场景化清洗策略¶

    1. 企业文档清洗¶

    文档类型 清洗重点
    合同 去除签署日期、盖章信息，保留条款
    报表 去除页眉页脚，保留表格数据
    政策文件 去除版本号、发布日期，保留正文
    邮件 去除签名档、引用的历史邮件

    2. 网页内容清洗¶

    元素 处理方式
    导航栏 识别并去除
    广告 去除
    评论区 保留或去除（视场景）
    Cookie 提示 去除
    正文 保留

    3. 技术文档清洗¶

    元素 处理方式
    代码块 保留，但去除行号
    API 文档 保留参数说明
    日志 去除时间戳、级别标识

    数据清洗最佳实践¶

    1. 可配置化¶

    // 清洗配置
type CleaningConfig struct {
    RemoveWatermarks    bool
    RemoveLineNumbers   bool
    NormalizeUnicode    bool
    ConvertFullwidth    bool
    RemoveHtmlTags      bool
    MinContentLength    int // 最小内容长度，过短则丢弃
}

func DefaultCleaningConfig() *CleaningConfig {
    return &CleaningConfig{
        RemoveWatermarks:  true,
        RemoveLineNumbers: true,
        NormalizeUnicode:  true,
        ConvertFullwidth:  true,
        RemoveHtmlTags:    true,
        MinContentLength:  10,
    }
}

    2. 质量验证¶

    // 清洗后质量验证
type CleaningValidator struct {
    minLength int
    maxLength int
}

func (v *CleaningValidator) Validate(text string) (bool, string) {
    // 检查长度
    if len(text) < v.minLength {
        return false, fmt.Sprintf("内容过短: %d 字符", len(text))
    }
    if len(text) > v.maxLength {
        return false, fmt.Sprintf("内容过长: %d 字符", len(text))
    }

    // 检查噪音字符比例
    noiseRatio := countNoiseRatio(text)
    if noiseRatio > 0.3 {
        return false, fmt.Sprintf("噪音比例过高: %.1f%%", noiseRatio*100)
    }

    return true, ""
}

    3. 调试和日志¶

    // 清洗过程日志
func (p *CleaningPipeline) ExecuteWithLog(text string) (string, []string) {
    result := text
    logs := []string{}

    for i, step := range p.steps {
        before := len(result)
        result = step(result)
        after := len(result)

        logs = append(logs, fmt.Sprintf("Step %d: %T - %d -> %d 字符",
            i+1, step, before, after))
    }

    return result, logs
}

    总结¶

    清洗类型 作用
    去噪 去除 HTML 残留、特殊字符
    规范化 繁简转换、全半角、大小写
    去格式化 去除链接、页眉页脚
    分块优化 规范化段落、句子边界

    数据清洗是 RAG 质量的重要保障，建议根据实际文档类型配置清洗策略。
